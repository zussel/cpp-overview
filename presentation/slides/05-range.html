Range basierte for loops

Bisher aufwendige Schreibweise


std::vector<int> v = {0, 1, 2, 3, 4, 5};

for (std::vector<int>::iterator i = v.begin(); i != v.end(); ++i {
    std::cout << i << "\n";
}

Die neue Schreibweise ist einfacher


std::vector<int> v = {0, 1, 2, 3, 4, 5};

for (const int&i : v) {
    std::cout << i << "\n";
}
